# 简单文本分类
* 数据范围：**两分类**的内容
  * 标签或者特征只有两类，对应数值未`0`和`1`
  * 数据处理逻辑
    * 去重无效内容
    * 构建分词表
    * 构建标签表(只有两分类，故而是固定的)
* 模型定义：`特征提取 → 加非线性 → 输出分数 → 转概率`
  * **词嵌入** `nn.Embedding`:将单词翻译为向量(机器可读)
  * **池化聚合** `nn.AdaptiveAvgPool1d`: 将一整句话的向量平均
    * 不论所有句子多长，都定义平均值的向量长度，不足的补齐(默认0)
  * **全连接分类** `nn.Sequential`: 把句子向量(**输入**)映射成分类结果(**输出**)
* 损失函数：**二分类损失**`BCELoss`
```py
import torch
import torch.nn as nn
import torch.optim as optim
from torch.utils.data import Dataset, DataLoader

# ====================== 1. 定义小型数据集 ======================
# 我们自己造一个超小情感分类数据集：正面/负面 短句
small_text_dataset = [
    ("我很喜欢这个产品", "正面"),
    ("用起来非常棒", "正面"),
    ("推荐大家购买", "正面"),
    ("质量太差了", "负面"),
    ("很不好用", "负面"),
    ("失望透顶", "负面"),
    ("效果很好很满意", "正面"),
    ("完全不值得买", "负面")
]

# 标签映射
label2idx = {"正面": 1, "负面": 0}
idx2label = {1: "正面", 0: "负面"}

# ====================== 2. 数据集解析类（核心！） ======================
# 作用：把原始文本 → 模型能看懂的数字张量
class TextDataset(Dataset):
    def __init__(self, data):
        self.data = data
        
        # 构建词表（从数据中提取所有词）
        self.vocab = {"<UNK>": 0}  # 未知词
        for text, _ in data:
            for word in text:  # 这里用单字分词（最简单）
                if word not in self.vocab:
                    self.vocab[word] = len(self.vocab)
        self.vocab_size = len(self.vocab)

    def __len__(self):
        return len(self.data)

    # 核心：文本 → 数字索引（模型只认识数字）
    def text_to_idx(self, text):
        return [self.vocab.get(word, 0) for word in text]

    # 取单条数据
    def __getitem__(self, idx):
        text, label = self.data[idx]
        text_idx = self.text_to_idx(text)
        label_idx = label2idx[label]
        
        # 转张量
        return torch.tensor(text_idx, dtype=torch.long), torch.tensor(label_idx, dtype=torch.float)

# 批处理函数（处理变长文本）
def collate_fn(batch):
    texts, labels = zip(*batch)
    texts = nn.utils.rnn.pad_sequence(texts, batch_first=True)  # 补零对齐
    labels = torch.stack(labels)
    return texts, labels

# ====================== 3. 初始化数据集 & 数据加载器 ======================
dataset = TextDataset(small_text_dataset)
dataloader = DataLoader(dataset, batch_size=2, shuffle=True, collate_fn=collate_fn)

# 查看数据集解析结果
print("词表大小:", dataset.vocab_size)
print("词表示例:", dict(list(dataset.vocab.items())[:10]))
print("="*50)

# ====================== 4. 构建小型文本模型 ======================
class TinyTextModel(nn.Module):
    def __init__(self, vocab_size, embed_dim=16, hidden_dim=8):
        super().__init__()
        # 1. 词嵌入：把数字 → 向量
        self.embedding = nn.Embedding(vocab_size, embed_dim)
        # 2. 池化：把整句话压缩成一个向量
        self.pool = nn.AdaptiveAvgPool1d(1)
        # 3. 分类头：输出预测结果
        self.fc = nn.Sequential(
            # 特征提取 → 加非线性 → 输出分数 → 转概率
            nn.Linear(embed_dim, hidden_dim),
            nn.ReLU(),
            nn.Linear(hidden_dim, 1),
            nn.Sigmoid()
        )

    def forward(self, x):
        # 前向传播流程
        x = self.embedding(x)  # [batch, seq_len] → [batch, seq_len, embed_dim]
        x = x.transpose(1, 2)  # 适配池化层
        x = self.pool(x).squeeze(-1)  # [batch, embed_dim]
        x = self.fc(x)  # [batch, 1]
        return x

# 初始化模型
model = TinyTextModel(vocab_size=dataset.vocab_size)
print("模型结构：",model)
print("模型参数：",len(list(model.parameters())))
for param in model.parameters():
    print(param)
print("="*50)

# ====================== 5. 训练配置 ======================
criterion = nn.BCELoss()  # 二分类损失
optimizer = optim.Adam(model.parameters(), lr=0.01)
epochs = 20

# ====================== 6. 训练流程（核心！） ======================
print("开始训练...")
for epoch in range(epochs):
    total_loss = 0
    for texts, labels in dataloader:
        # 1. 梯度清零
        optimizer.zero_grad()
        
        # 2. 前向传播
        outputs = model(texts).squeeze()
        
        # 3. 计算损失
        loss = criterion(outputs, labels)
        
        # 4. 反向传播 + 更新参数
        loss.backward()
        optimizer.step()
        
        total_loss += loss.item()

    # 打印进度
    if (epoch+1) % 5 == 0:
        print(f"Epoch [{epoch+1}/{epochs}], Loss: {total_loss:.4f}")

print("训练完成！")
print("="*50)

# ====================== 7. 预测流程 ======================
def predict(text):
    model.eval()
    with torch.no_grad():
        text_idx = [dataset.vocab.get(word, 0) for word in text]
        text_tensor = torch.tensor([text_idx], dtype=torch.long)
        pred = model(text_tensor).item()
        label = idx2label[round(pred)]
        score = pred if label == "正面" else 1-pred
    return label, score

# 测试
test_texts = ["这个产品很好用", "质量很差不要买"]
for text in test_texts:
    label, score = predict(text)
    print(f"输入：{text} → 预测：{label}（置信度：{score:.2f}）")
```

# 多标签文本分类
* 数据格式: `6552479093923774734_!_104_!_news_finance_!_目前有35万存款，怎样在短期之内翻倍？_!_`
  * `{content_id}_!_{label_id}_!_{label}_!_{content}`
  * 数据处理逻辑
    * 构建分词表
    * 构建标签表
* 模型定义
  * 
```py
import torch 
import torch.nn as nn
from torch.utils.data import DataLoader, Dataset

class DongleDataset(Dataset):
    def __init__(self,path):
        content_list = self.load_data(path)
        self.data = []
        self.idx2label = {} # idx to label 
        self.label2idx = {} # label to idx
        self.vocab = {"<UNK>":0}
        
        # 数据格式：6552479093923774734_!_104_!_news_finance_!_目前有35万存款，怎样在短期之内翻倍？_!_
            
        labels = set()
        # handle text and label
        for line in content_list:  # 只取前100个样本进行测试
            contents = line.strip().split('_!_')
            label = contents[2]
            labels.add(contents[2])

            content = contents[3]
            self.data.append((content,label))
            for word in content:
                if word not in self.vocab:
                    self.vocab[word] = len(self.vocab)

        #   build label mapping
        for idx,label in enumerate(labels):
            self.label2idx[label] = idx
            self.idx2label[idx] = label
        
        # count vocab and labels
        self.vocab_size = len(self.vocab)
        self.num_labels = len(self.label2idx)

    def load_data(self,path):
        with open(path,'r',encoding="utf-8") as f:
            content = f.readlines()
        return content
    
    def __len__(self):
        return len(self.data)
    
    def __getitem__(self,idx):
        content,label = self.data[idx]
        content_idx = [self.vocab.get(word,0) for word in content]
        label_idx = self.label2idx[label]
        return torch.tensor(content_idx,dtype=torch.long),torch.tensor(label_idx,dtype=torch.long)

def collate_fn(batch):
    texts,labels = zip(*batch)
    texts = nn.utils.rnn.pad_sequence(texts, batch_first=True)  # 补零对齐
    labels = torch.tensor(labels,dtype=torch.long)
    return texts, labels

dataset = DongleDataset('data/toutiao_cat_data.txt')
dataloader = DataLoader(dataset,batch_size=10,shuffle=True,collate_fn=collate_fn)

print(f"data count:{len(dataset.data)},vocab size:{dataset.vocab_size},labels count:{len(dataset.label2idx)}")
print("="*50)

class DongleModel(nn.Module):
    def __init__(self,vocab_size,num_labels,embed_dim = 16,hidden_dim = 8):
        super().__init__()
        self.embedding = nn.Embedding(vocab_size,embed_dim)
        self.pool = nn.AdaptiveAvgPool1d(1)
        self.fc = nn.Sequential(
            nn.Linear(embed_dim,hidden_dim),
            nn.ReLU(),
            nn.Linear(hidden_dim,num_labels), # 输出类别数
        )
    
    def forward(self,x):
        x = self.embedding(x)
        x = x.transpose(1,2)
        x = self.pool(x).squeeze(-1)
        x = self.fc(x)
        return x

model = DongleModel(dataset.vocab_size,dataset.num_labels)
loss_fn = nn.CrossEntropyLoss()
optimizer = torch.optim.Adam(model.parameters(),lr=0.001)

def train():
    print("Start model train.....")
    epochs = 20
    for epoch in range(epochs):
        total_loss = 0
        model.train()
        for contents,labels in dataloader:
            optimizer.zero_grad()
            outputs = model(contents)
            loss = loss_fn(outputs,labels)
            loss.backward()
            optimizer.step()
            total_loss += loss.item()
        print(f"Epoch {epoch+1}/{epochs}, Loss: {total_loss/len(dataloader):.4f}")

    print("Training completed!")
    print("="*50)

torch.save(model,"data/models/dongle_model.pth")

# 测试模型
def predict(test_content):
    model.eval()
    with torch.no_grad():
        c_idx = [dataset.vocab.get(word,0) for word in test_content]
        tensor = torch.tensor([c_idx],dtype=torch.long) # 增加batch维度
        output = model(tensor)
        pred_idx = torch.argmax(output).item() # 取最大概率类别
        pred_label = dataset.idx2label[pred_idx]
    return pred_idx,pred_label

test_content = "目前有35万存款，怎样在短期之内翻倍？"
predicted_label_idx,predicted_label_name = predict(test_content)
print(f"Test content: {test_content}, Predicted label index: {predicted_label_idx}, label : {predicted_label_name}")
```
